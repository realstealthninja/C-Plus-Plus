8 *
9 * \details
10 * Depth First Search also quoted as DFS is a Graph Traversal Algorithm.
11 * Time Complexity O(|V| + |E|) where V is number of vertices and E
12 * is number of edges in graph.
13 *
14 * Application of Depth First Search are
15 *
16 * 1. Finding connected components
17 * 2. Finding 2-(edge or vertex)-connected components.
18 * 3. Finding 3-(edge or vertex)-connected components.
19 * 4. Finding the bridges of a graph.
20 * 5. Generating words in order to plot the limit set of a group.
21 * 6. Finding strongly connected components.
22 *
23 * And there are many more...
24 *
25 * <h4>Working</h4>
26 * 1. Mark all vertices as unvisited first
27 * 2. start exploring from some starting vertex.
28 *
29 * While exploring vertex we mark the vertex as visited
30 * and start exploring the vertices connected to this
31 * vertex in recursive way.
32 *
33 */

46/**

53 * @param v second vertex

56void addEdge(std::vector<std::vector<size_t>> *adj, size_t u, size_t v) {
57 /*
58 *
59 * Here we are considering undirected graph that's the
60 * reason we are adding v to the adjacency list representation of u
61 * and also adding u to the adjacency list representation of v
62 *
63 */
64 (*adj)[u - 1].push_back(v - 1);
65 (*adj)[v - 1].push_back(u - 1);
66}

82 std::cout << v + 1 << " ";

99void depth_first_search(const std::vector<std::vector<size_t>> &adj,
100 size_t start) {
101 size_t vertices = adj.size();
102
103 std::vector<bool> visited(vertices, false);
