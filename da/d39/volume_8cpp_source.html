22/**
23 * @brief The volume of a [cube](https://en.wikipedia.org/wiki/Cube)
24 * @param length The length of the cube
25 * @returns The volume of the cube
26 */
27template <typename T>
28T cube_volume(T length) {
29 return std::pow(length, 3);
30}
31
32/**
33 * @brief The volume of a
34 * [rectangular](https://en.wikipedia.org/wiki/Cuboid) prism
35 * @param length The length of the base rectangle
36 * @param width The width of the base rectangle
37 * @param height The height of the rectangular prism
38 * @returns The volume of the rectangular prism
39 */
40template <typename T>
41T rect_prism_volume(T length, T width, T height) {
42 return length * width * height;
43}
44
45/**
46 * @brief The volume of a [cone](https://en.wikipedia.org/wiki/Cone)
47 * @param radius The radius of the base circle
48 * @param height The height of the cone
49 * @param PI The definition of the constant PI
50 * @returns The volume of the cone
51 */
52template <typename T>
53T cone_volume(T radius, T height, double PI = 3.14) {
54 return std::pow(radius, 2) * PI * height / 3;
55}
56
57/**
58 * @brief The volume of a
59 * [triangular](https://en.wikipedia.org/wiki/Triangular_prism) prism
60 * @param base The length of the base triangle
61 * @param height The height of the base triangles
62 * @param depth The depth of the triangular prism (the height of the whole
63 * prism)
64 * @returns The volume of the triangular prism
65 */
66template <typename T>
67T triangle_prism_volume(T base, T height, T depth) {
68 return base * height * depth / 2;
69}
70
71/**
72 * @brief The volume of a
73 * [pyramid](https://en.wikipedia.org/wiki/Pyramid_(geometry))
74 * @param length The length of the base shape (or base for triangles)
75 * @param width The width of the base shape (or height for triangles)
76 * @param height The height of the pyramid
77 * @returns The volume of the pyramid
78 */
79template <typename T>
80T pyramid_volume(T length, T width, T height) {
81 return length * width * height / 3;
82}
83
84/**
85 * @brief The volume of a [sphere](https://en.wikipedia.org/wiki/Sphere)
86 * @param radius The radius of the sphere
87 * @param PI The definition of the constant PI
88 * @returns The volume of the sphere
89 */
90template <typename T>
91T sphere_volume(T radius, double PI = 3.14) {
92 return PI * std::pow(radius, 3) * 4 / 3;
93}
94
95/**
96 * @brief The volume of a [cylinder](https://en.wikipedia.org/wiki/Cylinder)
97 * @param radius The radius of the base circle
98 * @param height The height of the cylinder
99 * @param PI The definition of the constant PI
100 * @returns The volume of the cylinder
101 */
102template <typename T>
103T cylinder_volume(T radius, T height, double PI = 3.14) {
104 return PI * std::pow(radius, 2) * height;
105}
