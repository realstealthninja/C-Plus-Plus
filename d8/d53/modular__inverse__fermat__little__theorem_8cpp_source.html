52int64_t binExpo(int64_t a, int64_t b, int64_t m) {
53 a %= m;
54 int64_t res = 1;
55 while (b > 0) {
56 if (b % 2) {
57 res = res * a % m;
58 }
59 a = a * a % m;
60 // Dividing b by 2 is similar to right shift.
61 b >>= 1;
62 }
63 return res;
64}
