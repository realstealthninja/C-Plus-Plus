1/**
2 * \file
3 * \authors [Krishna Vedala](https://github.com/kvedala)
4 * \brief Solve a multivariable first order [ordinary differential equation
5 * (ODEs)](https://en.wikipedia.org/wiki/Ordinary_differential_equation) using
6 * [forward Euler
7 * method](https://en.wikipedia.org/wiki/Numerical_methods_for_ordinary_differential_equations#Euler_method)
8 *
9 * \details
10 * The ODE being solved is:
11 * \f{eqnarray*}{
12 * \dot{u} &=& v\\
13 * \dot{v} &=& -\omega^2 u\\
14 * \omega &=& 1\\
15 * [x_0, u_0, v_0] &=& [0,1,0]\qquad\ldots\text{(initial values)}
16 * \f}
17 * The exact solution for the above problem is:
18 * \f{eqnarray*}{
19 * u(x) &=& \cos(x)\\
20 * v(x) &=& -\sin(x)\\
21 * \f}
22 * The computation results are stored to a text file `forward_euler.csv` and the
23 * exact soltuion results in `exact.csv` for comparison.
24 * <img
25 * src="https://raw.githubusercontent.com/TheAlgorithms/C-Plus-Plus/docs/images/numerical_methods/ode_forward_euler.svg"
26 * alt="Implementation solution"/>
27 *
28 * To implement [Van der Pol
29 * oscillator](https://en.wikipedia.org/wiki/Van_der_Pol_oscillator), change the
30 * ::problem function to:
31 * ```cpp
32 * const double mu = 2.0;
33 * dy[0] = y[1];
34 * dy[1] = mu * (1.f - y[0] * y[0]) * y[1] - y[0];
35 * ```
36 * \see ode_midpoint_euler.cpp, ode_semi_implicit_euler.cpp
37 */

45/**
46 * @brief Problem statement for a system with first-order differential
47 * equations. Updates the system differential variables.
48 * \note This function can be updated to and ode of any order.
49 *
50 * @param[in] x independent variable(s)
51 * @param[in,out] y dependent variable(s)
52 * @param[in,out] dy first-derivative of dependent variable(s)
53 */
54void problem(const double &x, std::valarray<double> *y,
55 std::valarray<double> *dy) {
56 const double omega = 1.F; // some const for the problem
57 (*dy)[0] = (*y)[1]; // x dot // NOLINT
58 (*dy)[1] = -omega * omega * (*y)[0]; // y dot // NOLINT
59}

61/**
62 * @brief Exact solution of the problem. Used for solution comparison.
63 *
64 * @param[in] x independent variable
65 * @param[in,out] y dependent variable
66 */
67void exact_solution(const double &x, std::valarray<double> *y) {
68 y[0][0] = std::cos(x);
69 y[0][1] = -std::sin(x);
70}

72/** \addtogroup ode Ordinary Differential Equations
73 * Integration functions for implementations with solving [ordinary differential
74 * equations](https://en.wikipedia.org/wiki/Ordinary_differential_equation)
75 * (ODEs) of any order and and any number of independent variables.
76 * @{
77 */
78/**
79 * @brief Compute next step approximation using the forward-Euler
80 * method. @f[y_{n+1}=y_n + dx\cdot f\left(x_n,y_n\right)@f]
81 * @param[in] dx step size
82 * @param[in] x take \f$x_n\f$ and compute \f$x_{n+1}\f$
83 * @param[in,out] y take \f$y_n\f$ and compute \f$y_{n+1}\f$
84 * @param[in,out] dy compute \f$f\left(x_n,y_n\right)\f$
85 */
86void forward_euler_step(const double dx, const double x,
87 std::valarray<double> *y, std::valarray<double> *dy) {
88 problem(x, y, dy);
89 *y += *dy * dx;
90}

92/**
93 * @brief Compute approximation using the forward-Euler
94 * method in the given limits.
95 * @param[in] dx step size
96 * @param[in] x0 initial value of independent variable
97 * @param[in] x_max final value of independent variable
98 * @param[in,out] y take \f$y_n\f$ and compute \f$y_{n+1}\f$
99 * @param[in] save_to_file flag to save results to a CSV file (1) or not (0)
100 * @returns time taken for computation in seconds
101 */
102double forward_euler(double dx, double x0, double x_max,
103 std::valarray<double> *y, bool save_to_file = false) {
104 std::valarray<double> dy = *y;
105
106 std::ofstream fp;
107 if (save_to_file) {
108 fp.open("forward_euler.csv", std::ofstream::out);
109 if (!fp.is_open()) {
110 std::perror("Error! ");
111 }
112 }
113
114 std::size_t L = y->size();
115
116 /* start integration */
117 std::clock_t t1 = std::clock();
118 double x = x0;
119
120 do { // iterate for each step of independent variable
121 if (save_to_file && fp.is_open()) {
122 // write to file
123 fp << x << ",";
124 for (int i = 0; i < L - 1; i++) {
125 fp << y[0][i] << ","; // NOLINT
126 }
127 fp << y[0][L - 1] << "\n"; // NOLINT
128 }
129
130 forward_euler_step(dx, x, y, &dy); // perform integration
131 x += dx; // update step
132 } while (x <= x_max); // till upper limit of independent variable
133 /* end of integration */
134 std::clock_t t2 = std::clock();
135
136 if (fp.is_open()) {
137 fp.close();
138 }
139
140 return static_cast<double>(t2 - t1) / CLOCKS_PER_SEC;
141}

161 std::perror("Error! ");
162 return;

164 std::cout << "Finding exact solution\n";
165
166 std::clock_t t1 = std::clock();
167 do {
168 fp << x << ",";
169 for (int i = 0; i < y.size() - 1; i++) {
170 fp << y[i] << ","; // NOLINT
171 }
172 fp << y[y.size() - 1] << "\n"; // NOLINT
173
174 exact_solution(x, &y);
175
176 x += step_size;
177 } while (x <= X_MAX);
178
179 std::clock_t t2 = std::clock();
180 double total_time = static_cast<double>(t2 - t1) / CLOCKS_PER_SEC;
181 std::cout << "\tTime = " << total_time << " ms\n";

183 fp.close();
184}
