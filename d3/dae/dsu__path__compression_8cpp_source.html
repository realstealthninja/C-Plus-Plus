2 * @file
3 * @brief [DSU (Disjoint
4 * sets)](https://en.wikipedia.org/wiki/Disjoint-set-data_structure)
5 * @details
6 * It is a very powerful data structure that keeps track of different
7 * clusters(sets) of elements, these sets are disjoint(doesnot have a common
8 * element). Disjoint sets uses cases : for finding connected components in a
9 * graph, used in Kruskal's algorithm for finding Minimum Spanning tree.
10 * Operations that can be performed:
11 * 1) UnionSet(i,j): add(element i and j to the set)
12 * 2) findSet(i): returns the representative of the set to which i belogngs to.
13 * 3) get_max(i),get_min(i) : returns the maximum and minimum
14 * Below is the class-based approach which uses the heuristic of path
15 * compression. Using path compression in findSet(i),we are able to get to the
16 * representative of i in O(1) time.
17 * @author [AayushVyasKIIT](https://github.com/AayushVyasKIIT)
18 * @see dsu_union_rank.cpp
19 */

29/**
30 * @brief Disjoint sets union data structure, class based representation.
31 * @param n number of elements
32 */
33class dsu {
34 private:
35 vector<uint64_t> p; ///< keeps track of the parent of ith element
36 vector<uint64_t> depth; ///< tracks the depth(rank) of i in the tree
37 vector<uint64_t> setSize; ///< size of each chunk(set)
38 vector<uint64_t> maxElement; ///< maximum of each set to which i belongs to
39 vector<uint64_t> minElement; ///< minimum of each set to which i belongs to
40 public:
41 /**
42 * @brief contructor for initialising all data members.
43 * @param n number of elements
44 */
45 explicit dsu(uint64_t n) {
46 p.assign(n, 0);
47 /// initially, all of them are their own parents
48 for (uint64_t i = 0; i < n; i++) {
49 p[i] = i;
50 }
51 /// initially all have depth are equals to zero
52 depth.assign(n, 0);
53 maxElement.assign(n, 0);
54 minElement.assign(n, 0);
55 for (uint64_t i = 0; i < n; i++) {
56 depth[i] = 0;
57 maxElement[i] = i;
58 minElement[i] = i;
59 }
60 setSize.assign(n, 0);
61 /// initially set size will be equals to one
62 for (uint64_t i = 0; i < n; i++) {
63 setSize[i] = 1;
64 }
65 }
66
67 /**
68 * @brief Method to find the representative of the set to which i belongs
69 * to, T(n) = O(1)
70 * @param i element of some set
71 * @returns representative of the set to which i belongs to.
72 */
73 uint64_t findSet(uint64_t i) {
74 /// using path compression
75 if (p[i] == i) {
76 return i;
77 }
78 return (p[i] = findSet(p[i]));
79 }
80 /**
81 * @brief Method that combines two disjoint sets to which i and j belongs to
82 * and make a single set having a common representative.
83 * @param i element of some set
84 * @param j element of some set
85 * @returns void
86 */
87 void UnionSet(uint64_t i, uint64_t j) {
88 /// check if both belongs to the same set or not
89 if (isSame(i, j)) {
90 return;
91 }
92
93 // we find the representative of the i and j
94 uint64_t x = findSet(i);
95 uint64_t y = findSet(j);
96
97 /// always keeping the min as x
98 /// shallow tree
99 if (depth[x] > depth[y]) {
100 std::swap(x, y);
101 }
102 /// making the shallower root's parent the deeper root
103 p[x] = y;
104
105 /// if same depth, then increase one's depth
106 if (depth[x] == depth[y]) {
107 depth[y]++;
108 }
109 /// total size of the resultant set
110 setSize[y] += setSize[x];
111 /// changing the maximum elements
112 maxElement[y] = std::max(maxElement[x], maxElement[y]);
113 minElement[y] = std::min(minElement[x], minElement[y]);
114 }
115 /**
116 * @brief A utility function which check whether i and j belongs to
117 * same set or not
118 * @param i element of some set
119 * @param j element of some set
120 * @returns `true` if element `i` and `j` ARE in the same set
121 * @returns `false` if element `i` and `j` are NOT in same set
122 */
123 bool isSame(uint64_t i, uint64_t j) {
124 if (findSet(i) == findSet(j)) {
125 return true;
126 }
127 return false;
128 }
129 /**
130 * @brief prints the minimum, maximum and size of the set to which i belongs
131 * to
132 * @param i element of some set
133 * @returns void
134 */
135 vector<uint64_t> get(uint64_t i) {
136 vector<uint64_t> ans;
137 ans.push_back(get_min(i));
138 ans.push_back(get_max(i));
139 ans.push_back(size(i));
140 return ans;
141 }
142 /**
143 * @brief A utility function that returns the size of the set to which i
144 * belongs to
145 * @param i element of some set
146 * @returns size of the set to which i belongs to
147 */
148 uint64_t size(uint64_t i) { return setSize[findSet(i)]; }
149 /**
150 * @brief A utility function that returns the max element of the set to
151 * which i belongs to
152 * @param i element of some set
153 * @returns maximum of the set to which i belongs to
154 */
155 uint64_t get_max(uint64_t i) { return maxElement[findSet(i)]; }
156 /**
157 * @brief A utility function that returns the min element of the set to
158 * which i belongs to
159 * @param i element of some set
160 * @returns minimum of the set to which i belongs to
161 */
162 uint64_t get_min(uint64_t i) { return minElement[findSet(i)]; }
163};
