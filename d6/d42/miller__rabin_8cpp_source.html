13/**
14 * Function to give a binary representation of a number in reverse order
15 * @param num integer number that we want to convert
16 * @return result vector of the number input in reverse binary
17 */
18template <typename T>
19std::vector<T> reverse_binary(T num) {
20 std::vector<T> result;
21 T temp = num;
22 while (temp > 0) {
23 result.push_back(temp % 2);
24 temp = temp / 2;
25 }
26 return result;
27}

29/**
30 * Function for modular exponentiation.
31 * This function is an efficient modular exponentiation function.
32 * It can be used with any big integer library such as Boost multiprecision
33 * to give result any modular exponentiation problem relatively quickly.
34 * @param base number being raised to a power as integer
35 * @param rev_binary_exponent reverse binary of the power the base is being
36 * raised to
37 * @param mod modulo
38 * @return r the modular exponentiation of \f$a^{n} \equiv r \mod{m}\f$ where
39 * \f$n\f$ is the base 10 representation of rev_binary_exponent and \f$m = mod
40 * \f$ parameter.
41 */
42template <typename T>
43T modular_exponentiation(T base, const std::vector<T> &rev_binary_exponent,
44 T mod) {
45 if (mod == 1)
46 return 0;
47 T b = 1;
48 if (rev_binary_exponent.size() == 0)
49 return b;
50 T A = base;
51 if (rev_binary_exponent[0] == 1)
52 b = base;
53
54 for (typename std::vector<T>::const_iterator it =
55 rev_binary_exponent.cbegin() + 1;
56 it != rev_binary_exponent.cend(); ++it) {
57 A = A * A % mod;
58 if (*it == 1)
59 b = A * b % mod;
60 }
61 return b;
62}

64/** Function for testing the conditions that are satisfied when a number is
65 * prime.
66 * @param d number such that \f$d \cdot 2^r = n - 1\f$ where \f$n = num\f$
67 * parameter and \f$r \geq 1\f$
68 * @param num number being tested for primality.
69 * @return 'false' if n is composite
70 * @return 'true' if n is (probably) prime.
71 */
72template <typename T>
73bool miller_test(T d, T num) {
74 // random number seed
75 std::random_device rd_seed;
76 // random number generator
77 std::mt19937 gen(rd_seed());
78 // Uniformly distributed range [2, num - 2] for random numbers
79 std::uniform_int_distribution<> distribution(2, num - 2);
80 // Random number generated in the range [2, num -2].
81 T random = distribution(gen);
82 // vector for reverse binary of the power
83 std::vector<T> power = reverse_binary(d);
84 // x = random ^ d % num
85 T x = modular_exponentiation(random, power, num);
86 // miller conditions
87 if (x == 1 || x == num - 1) {
88 return true;
89 }
90
91 while (d != num - 1) {
92 x = (x * x) % num;
93 d *= 2;
94 if (x == 1) {
95 return false;
96 }
97 if (x == num - 1) {
98 return true;
99 }
100 }
101 return false;
102}

104/**
105 * Function that test (probabilistically) whether a given number is a prime
106 * based on the Miller-Rabin Primality Test.
107 * @param num number to be tested for primality.
108 * @param repeats number of repetitions for the test to increase probability of
109 * correct result.
110 * @return 'false' if num is composite
111 * @return 'true' if num is (probably) prime
112 *
113 * \detail
114 * First we check whether the num input is less than 4, if so we can determine
115 * whether this is a prime or composite by checking for 2 and 3.
116 * Next we check whether this num is odd (as all primes greater than 2 are odd).
117 * Next we write our num in the following format \f$num = 2^r \cdot d + 1\f$.
118 * After finding r and d for our input num, we use for loop repeat number of
119 * times inside which we check the miller conditions using the function
120 * miller_test. If miller_test returns false then the number is composite After
121 * the loop finishes completely without issuing a false return call, we can
122 * conclude that this number is probably prime.
123 */
124template <typename T>
125bool miller_rabin_primality_test(T num, T repeats) {
126 if (num <= 4) {
127 // If num == 2 or num == 3 then prime
128 if (num == 2 || num == 3) {
129 return true;
130 } else {
131 return false;
132 }
133 }
134 // If num is even then not prime
135 if (num % 2 == 0) {
136 return false;
137 }
138 // Finding d and r in num = 2^r * d + 1
139 T d = num - 1, r = 0;
140 while (d % 2 == 0) {
141 d = d / 2;
142 r++;
143 }
144
145 for (T i = 0; i < repeats; ++i) {
146 if (!miller_test(d, num)) {
147 return false;
148 }
149 }
150 return true;
151}
