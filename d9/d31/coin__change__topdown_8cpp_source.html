37/**
38 * @brief This implementation is for finding minimum number of coins .
39 * @param T template-type to use any kind of value
40 * @param n amount to be reached
41 * @param coins vector of coins
42 * @param t deontes the number of coins
43 * @param dp initilised to 0
44 * @returns minimum number of coins
45 */
46template <typename T>
47int64_t mincoins(const T &n, const std::vector<T> &coins, const int16_t &t,
48 std::vector<T> dp) {
49 if (n == 0) {
50 return 0;
51 }
52 if (dp[n] != 0) {
53 return dp[n];
54 }
55 int ans = INT_MAX; // variable to store min coins
56 for (int i = 0; i < t; i++) {
57 if (n - coins[i] >= 0) { // if after subtracting the current
58 // denomination is it greater than 0 or not
59 int sub = mincoins(n - coins[i], coins, t, dp);
60 ans = std::min(ans, sub + 1);
61 }
62 }
63 dp[n] = ans;
64 return dp[n]; // returns minimum number of coins
65}
