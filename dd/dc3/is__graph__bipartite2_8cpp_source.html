2 * @brief Check whether a given graph is bipartite or not
3 * @details
4 * A bipartite graph is the one whose nodes can be divided into two
5 * disjoint sets in such a way that the nodes in a set are not
6 * connected to each other at all, i.e. no intra-set connections.
7 * The only connections that exist are that of inter-set,
8 * i.e. the nodes from one set are connected to a subset of nodes
9 * in the other set.
10 * In this implementation, using a graph in the form of adjacency
11 * list, check whether the given graph is a bipartite or not.
12 *
13 * References used:
14 * [GeeksForGeeks](https://www.geeksforgeeks.org/bipartite-graph/)
15 * @author [tushar2407](https://github.com/tushar2407)
16 */

27/**
28 * @brief function to check whether the passed graph is bipartite or not
29 * @param graph is a 2D matrix whose rows or the first index signify the node
30 * and values in that row signify the nodes it is connected to
31 * @param index is the valus of the node currently under observation
32 * @param visited is the vector which stores whether a given node has been
33 * traversed or not yet
34 * @returns boolean
35 */
36bool checkBipartite(const std::vector<std::vector<int64_t>> &graph,
37 int64_t index, std::vector<int64_t> *visited) {
38 std::queue<int64_t> q; ///< stores the neighbouring node indexes in squence
39 /// of being reached
40 q.push(index); /// insert the current node into the queue
41 (*visited)[index] = 1; /// mark the current node as travelled
42 while (q.size()) {
43 int64_t u = q.front();
44 q.pop();
45 for (uint64_t i = 0; i < graph[u].size(); i++) {
46 int64_t v =
47 graph[u][i]; ///< stores the neighbour of the current node
48 if (!(*visited)[v]) /// check whether the neighbour node is
49 /// travelled already or not
50 {
51 (*visited)[v] =
52 ((*visited)[u] == 1)
53 ? -1
54 : 1; /// colour the neighbouring node with
55 /// different colour than the current node
56 q.push(v); /// insert the neighbouring node into the queue
57 } else if ((*visited)[v] ==
58 (*visited)[u]) /// if both the current node and its
59 /// neighbour has the same state then it
60 /// is not a bipartite graph
61 {
62 return false;
63 }
64 }
65 }
66 return true; /// return true when all the connected nodes of the current
67 /// nodes are travelled and satisify all the above conditions
68}
69/**
70 * @brief returns true if the given graph is bipartite else returns false
71 * @param graph is a 2D matrix whose rows or the first index signify the node
72 * and values in that row signify the nodes it is connected to
73 * @returns booleans
74 */
75bool isBipartite(const std::vector<std::vector<int64_t>> &graph) {
76 std::vector<int64_t> visited(
77 graph.size()); ///< stores boolean values
78 /// which signify whether that node had been visited or
79 /// not
80
81 for (uint64_t i = 0; i < graph.size(); i++) {
82 if (!visited[i]) /// if the current node is not visited then check
83 /// whether the sub-graph of that node is a bipartite
84 /// or not
85 {
86 if (!checkBipartite(graph, i, &visited)) {
87 return false;
88 }
89 }
90 }
91 return true;
92}
